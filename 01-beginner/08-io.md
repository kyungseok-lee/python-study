# 08. 입출력 (I/O)

## 🎯 학습 목표
- input() 으로 사용자 입력을 받는다
- print() 의 다양한 옵션을 활용한다
- 파일 읽기/쓰기의 기본을 안다

## 1. 표준 출력 (print)

### 1.1 기본 사용
```python
print("Hello, World!")
print(42)
print(3.14)
print([1, 2, 3])
```

### 1.2 여러 값 출력
```python
# 콤마로 구분 (자동으로 공백 추가)
print("a", "b", "c")        # a b c

# sep 인자로 구분자 변경
print("a", "b", "c", sep="-")    # a-b-c
print("a", "b", "c", sep="")     # abc
print("a", "b", "c", sep="\n")   # 줄바꿈

# end 인자로 끝 문자 변경 (기본은 \n)
print("Hello", end=" ")
print("World")              # Hello World

print("Loading", end="")
for i in range(3):
    print(".", end="", flush=True)
print(" Done!")             # Loading... Done!
```

### 1.3 파일에 출력
```python
with open("output.txt", "w") as f:
    print("Hello, File!", file=f)
```

### 1.4 stderr로 출력
```python
import sys
print("Error message", file=sys.stderr)
```

## 2. 표준 입력 (input)

### 2.1 기본 사용
```python
name = input("이름을 입력하세요: ")
print(f"안녕하세요, {name}님!")
```

> ⚠️ `input()` 은 **항상 문자열을 반환**합니다.

### 2.2 숫자 입력 받기
```python
# 정수
age = int(input("나이: "))
print(age + 1)

# 실수
height = float(input("키: "))
print(height * 100)
```

### 2.3 여러 값 입력
```python
# 공백으로 구분된 값
a, b = input("두 숫자: ").split()
a, b = int(a), int(b)

# 한 줄에 (map 활용)
a, b, c = map(int, input("세 숫자: ").split())

# 리스트로
numbers = list(map(int, input("숫자들: ").split()))
print(sum(numbers))
```

### 2.4 입력 검증
```python
while True:
    age_str = input("나이를 입력하세요: ")
    if age_str.isdigit():
        age = int(age_str)
        if 0 < age < 150:
            break
    print("올바른 나이를 입력하세요.")

print(f"입력된 나이: {age}")
```

### 2.5 비밀번호 입력 (입력 숨기기)
```python
import getpass

password = getpass.getpass("비밀번호: ")
print(f"비밀번호 길이: {len(password)}")
```

## 3. 파일 입출력 기본

### 3.1 파일 열기 모드
| 모드 | 설명 |
|------|------|
| `r` | 읽기 (기본) |
| `w` | 쓰기 (덮어쓰기) |
| `a` | 추가 (append) |
| `x` | 새 파일 생성 (있으면 에러) |
| `b` | 바이너리 모드 (예: `rb`) |
| `t` | 텍스트 모드 (기본) |
| `+` | 읽기/쓰기 (예: `r+`) |

### 3.2 with 문 사용 (권장 ✅)
```python
# 자동으로 파일이 닫힘
with open("file.txt", "r", encoding="utf-8") as f:
    content = f.read()
    print(content)
```

### 3.3 파일 쓰기
```python
# 새로 쓰기
with open("hello.txt", "w", encoding="utf-8") as f:
    f.write("Hello, World!\n")
    f.write("두 번째 줄\n")

# 여러 줄 한 번에
lines = ["첫 줄\n", "둘째 줄\n", "셋째 줄\n"]
with open("lines.txt", "w", encoding="utf-8") as f:
    f.writelines(lines)

# 추가 (append)
with open("hello.txt", "a", encoding="utf-8") as f:
    f.write("추가된 줄\n")
```

### 3.4 파일 읽기
```python
# 전체 읽기
with open("hello.txt", "r", encoding="utf-8") as f:
    content = f.read()
    print(content)

# 한 줄씩 읽기 (메모리 효율적)
with open("hello.txt", "r", encoding="utf-8") as f:
    for line in f:
        print(line.rstrip())  # 줄바꿈 제거

# 모든 줄을 리스트로
with open("hello.txt", "r", encoding="utf-8") as f:
    lines = f.readlines()
    print(lines)

# 한 줄만
with open("hello.txt", "r", encoding="utf-8") as f:
    first_line = f.readline()
    second_line = f.readline()
```

### 3.5 인코딩 주의
```python
# UTF-8 (권장 - 한글 포함 모든 문자)
with open("file.txt", "r", encoding="utf-8") as f:
    pass

# Windows 한글
with open("file.txt", "r", encoding="cp949") as f:
    pass

# 인코딩 미지정 시 시스템 기본 (Windows에서 한글 깨질 수 있음)
```

## 4. print 포맷팅 심화

### 4.1 정렬
```python
data = [
    ("Alice", 25, 165.5),
    ("Bob", 30, 175.2),
    ("Charlie", 22, 168.0),
]

print(f"{'이름':<10}{'나이':>5}{'키':>8}")
print("-" * 25)
for name, age, height in data:
    print(f"{name:<10}{age:>5}{height:>8.1f}")
```

출력:
```
이름        나이      키
-------------------------
Alice        25   165.5
Bob          30   175.2
Charlie      22   168.0
```

### 4.2 표 출력
```python
header = ["ID", "이름", "점수"]
rows = [
    [1, "Alice", 85],
    [2, "Bob", 92],
    [3, "Charlie", 78],
]

# 헤더
print(" | ".join(f"{h:^8}" for h in header))
print("-" * 30)

# 데이터
for row in rows:
    print(" | ".join(f"{str(c):^8}" for c in row))
```

### 4.3 진행 표시 (Progress)
```python
import time

n = 50
for i in range(n + 1):
    bar = "█" * (i // 2) + "-" * ((n - i) // 2)
    percent = i / n * 100
    print(f"\r[{bar}] {percent:.0f}%", end="", flush=True)
    time.sleep(0.05)
print()  # 줄바꿈
```

## 6. argparse (제대로 된 CLI)

```python
import argparse

parser = argparse.ArgumentParser(description="간단한 계산기")
parser.add_argument("a", type=int, help="첫 번째 숫자")
parser.add_argument("b", type=int, help="두 번째 숫자")
parser.add_argument("--op", choices=["add", "sub", "mul", "div"],
                    default="add", help="연산 종류")
parser.add_argument("-v", "--verbose", action="store_true",
                    help="자세한 출력")

args = parser.parse_args()

ops = {
    "add": lambda x, y: x + y,
    "sub": lambda x, y: x - y,
    "mul": lambda x, y: x * y,
    "div": lambda x, y: x / y,
}

result = ops[args.op](args.a, args.b)

if args.verbose:
    print(f"{args.a} {args.op} {args.b} = {result}")
else:
    print(result)
```

실행:
```bash
python3 calc.py 10 5 --op mul -v
# 10 mul 5 = 50

python3 calc.py --help
```

## 7. 파일 경로 처리 (pathlib)

```python
from pathlib import Path

# 현재 파일의 경로
current = Path(__file__)
print(current.absolute())   # 절대 경로

# 디렉토리 다루기
home = Path.home()          # 홈 디렉토리
cwd = Path.cwd()            # 현재 작업 디렉토리

# 경로 조합
file_path = Path("data") / "users.txt"
print(file_path)            # data/users.txt

# 파일 정보
p = Path("hello.txt")
print(p.exists())           # 존재 여부
print(p.is_file())          # 파일인가
print(p.is_dir())           # 디렉토리인가
print(p.name)               # 파일명
print(p.stem)               # 확장자 제외
print(p.suffix)             # 확장자
print(p.parent)             # 부모 디렉토리

# 디렉토리 내용
for item in Path(".").iterdir():
    print(item)

# 패턴 매칭
for py_file in Path(".").glob("*.py"):
    print(py_file)

# 재귀적으로
for py_file in Path(".").rglob("*.py"):
    print(py_file)

# 디렉토리 생성
Path("new_dir").mkdir(exist_ok=True)
Path("a/b/c").mkdir(parents=True, exist_ok=True)
```

### 파일 읽기/쓰기 (pathlib)
```python

# 텍스트

# 바이너리
p.write_bytes(b"\x00\x01\x02")

```

## 8. JSON 다루기 (미리보기)

```python
import json

# Python → JSON 파일
data = {"name": "Alice", "age": 25, "hobbies": ["reading", "coding"]}

with open("data.json", "w", encoding="utf-8") as f:
    json.dump(data, f, ensure_ascii=False, indent=2)

# JSON 파일 → Python
with open("data.json", "r", encoding="utf-8") as f:
    loaded = json.load(f)
    print(loaded)

# 문자열로 변환
json_str = json.dumps(data, ensure_ascii=False, indent=2)
parsed = json.loads(json_str)
```

## 9. CSV 다루기

```python
import csv

# 쓰기
with open("users.csv", "w", encoding="utf-8", newline="") as f:
    writer = csv.writer(f)
    writer.writerow(["이름", "나이", "도시"])
    writer.writerow(["Alice", 25, "Seoul"])
    writer.writerow(["Bob", 30, "Busan"])

# 읽기
with open("users.csv", "r", encoding="utf-8") as f:
    reader = csv.reader(f)
    for row in reader:
        print(row)

# 딕셔너리로 읽기
with open("users.csv", "r", encoding="utf-8") as f:
    reader = csv.DictReader(f)
    for row in reader:
        print(row["이름"], row["나이"])
```

## 10. 실전 예제

### 예제 1: 메모장
```python
def save_note():
    note = input("메모 입력: ")
    with open("notes.txt", "a", encoding="utf-8") as f:
        f.write(note + "\n")
    print("저장 완료!")

def show_notes():
    try:
        with open("notes.txt", "r", encoding="utf-8") as f:
            print("=== 메모 목록 ===")
            for i, line in enumerate(f, 1):
                print(f"{i}. {line.strip()}")
    except FileNotFoundError:
        print("메모가 없습니다.")

while True:
    cmd = input("\n[s]ave / [v]iew / [q]uit: ").lower()
    if cmd == "s":
        save_note()
    elif cmd == "v":
        show_notes()
    elif cmd == "q":
        break
```

### 예제 2: 파일 단어 카운트
```python
from collections import Counter

def word_count(filename):
    with open(filename, "r", encoding="utf-8") as f:
        text = f.read()

    words = text.lower().split()
    counter = Counter(words)

    print(f"총 단어 수: {sum(counter.values())}")
    print(f"고유 단어 수: {len(counter)}")
    print("\n빈도 Top 10:")
    for word, count in counter.most_common(10):
        print(f"  {word}: {count}")

# word_count("article.txt")
```

### 예제 3: TODO 리스트
```python
import json
from pathlib import Path

TODO_FILE = Path("todos.json")

def load_todos():
    if TODO_FILE.exists():
        return json.loads(TODO_FILE.read_text(encoding="utf-8"))
    return []

def save_todos(todos):
    TODO_FILE.write_text(
        json.dumps(todos, ensure_ascii=False, indent=2),
        encoding="utf-8"
    )

def add_todo(todos, task):
    todos.append({"task": task, "done": False})
    save_todos(todos)

def list_todos(todos):
    for i, t in enumerate(todos, 1):
        mark = "✓" if t["done"] else " "
        print(f"[{mark}] {i}. {t['task']}")

# 사용
todos = load_todos()
add_todo(todos, "Python 학습")
add_todo(todos, "운동하기")
list_todos(todos)
```
